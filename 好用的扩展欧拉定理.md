### 好用的扩展欧拉定理

```c++
#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
LL a,m;
string s;

LL qpow(LL a, LL b , LL p)
{
    LL ret = 1;
    while(b)
    {
        if(b&1) ret = ret * a % p;
        b>>=1;
        a = a*a%p;
    }
    return ret;
}

LL get_phi(LL x)
{
    LL ret = x;
    for(int i = 2 ; i<=x/i ; i++)
    {
        if(x % i == 0)
        {
            ret = ret * (i-1) / i;
            while(x%i == 0) x/=i;
        }
    }
    if(x>1) ret = ret / x * (x-1);
    return ret ;
}

//最后如果超过了 p 要补上
LL get_b(string& s , LL p)
{
    LL ret = 0;
    bool flag = false;
    for(auto t:s)
    {
        ret = ret * 10 + (t - '0');
        if(ret >= p)
        {
            ret %= p;
            flag = true;
        }
    }
    if(flag) ret += p;
    return ret;
}

int main()
{
    cin >> a >> m >> s;
    LL phi = get_phi(m);
    LL b = get_b(s , phi);
    LL ret = qpow(a , b , m);
    cout << ret << endl;
    
    return 0;
}
```
