```c++
#include<bits/stdc++.h>
using namespace std;
int n;
const int N = 1010;
int m[N][N];
bool st[N][N];
int dx[] = {0,0,1,-1,1,1,-1,-1};
int dy[] = {1,-1,0,0,1,-1,1,-1};
 bool ba,lo;

void dfs(int x , int y)
{
    if(st[x][y]) return;
    st[x][y] = true;
    for(int i = 0 ; i<8 ; i++)
    {
        int nx = x+dx[i];
        int ny = y+dy[i];
        if(nx>n||ny>n||nx<1||ny<1)
        {
            continue;
        }
        if(m[nx][ny]==m[x][y]) dfs(nx , ny);
        if(m[nx][ny]>m[x][y]) lo = true;
        if(m[nx][ny]<m[x][y]) ba = true;
        
    }
}

int main()
{
    cin >>n ; 
    for(int i = 1 ; i<=n ; i++)
    {
        for(int j = 1 ; j<=n ; j++)
        {
            cin >> m[i][j] ;
            
        }
    }
    bool flag = true;
    int a = m[1][1];
    for(int i = 1 ; i<=n ; i++)
    {
        for(int j = 1  ;j<=n ; j++)
        {
            if(a!=m[i][j]) flag = false;
        }
    }
    if(flag )
    {
        cout << 1 << " " << 1 << endl;
        return 0;
    }
   int rm = 0 , rl = 0;
    for(int i = 1 ;i<=n ; i++)
    {
        for(int j = 1 ; j<=n ; j++)
        {
            ba = false;
            lo = false;
            if(st[i][j]) continue;
            dfs(i , j);
            if(ba&&lo) continue;
            else if(ba) rm++;
            else rl++;
        }
    }
    cout << rm << " " << rl << endl;
    return 0;
}
```
