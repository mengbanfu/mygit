```c++
//区间查询还没更，懒得更了，下次遇到正好复习

#include<bits/stdc++.h>
using namespace std;
#define lc p<<1
#define rc p<<1 | 1
typedef long long LL;
const int N = 5e5+10;
LL a[N];
int n,m;

struct node{
    int l,r;
    LL sum ; LL add;
} tr[N<<2];

void lazy(int p , LL k)
{
    int l = tr[p].l ; int r = tr[p].r;
    tr[p].sum += (r - l + 1) * k;
    tr[p].add += k;
}

void pushdown(int p)
{
    lazy(lc , tr[p].add);
    lazy(rc , tr[p].add);
    tr[p].add = 0;
}


void pushup(int p)
{
    tr[p].sum = tr[lc].sum + tr[rc].sum; 
}

void build(int p , int l , int r)
{
    tr[p] = {l ,r , a[l] , 0};
    if(l==r) return;

    int mid = (l+r)>>1;
    build(lc , l , mid);
    build(rc , mid+1 , r);
    pushup(p);
}

void modify(int p , int x , int y , LL k)
{
    int l = tr[p].l ; int r = tr[p].r;
    if(x<=l && y>=r)
    {
        lazy(p , k);
        return ;
    }

    pushdown(p);
    int mid = (l+r)>>1;
    if(x<=mid) modify(lc , x , y , k);
    if(y>mid) modify(rc , x , y , k);
    pushup(p);
}

//单点查询
LL query(int p , int x  )
{
    int l = tr[p].l ; int r = tr[p].r;
    if(l==x && x==r)
    {
        return tr[p].sum;
    }
    pushdown(p);
    int mid = (l+r) >> 1;
    if(x<=mid) return query(lc , x );
    else return query(rc , x );
}

int main()
{
    cin >> n >> m;
    for(int i = 1 ; i<=n ; i++) cin >> a[i];

    build(1 , 1 , n);
    while(m--)
    {
        int op,x,y;LL k;
        cin >> op;
        if(op==1)
        {
            cin >> x >> y >> k;
            modify(1 , x , y , k);
        }
        if(op ==2)
        {
            cin >> x;
            cout << query(1 , x) <<endl; 
        }
    }
    return 0;
}
```
