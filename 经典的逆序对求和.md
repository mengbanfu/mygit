### 经典的逆序对求和

```c++
#include <iostream>
using namespace std;

const int N = 1e5 + 10;
int a[N], b[N], pos[N];
int tmp[N];
long long res;  // 答案会爆 int

// 归并排序求逆序对
void merge_sort(int l, int r, int q[]) {
    if (l >= r) return;
    int mid = l + r >> 1;
    merge_sort(l, mid, q);
    merge_sort(mid + 1, r, q);

    int i = l, j = mid + 1, k = 0;
    while (i <= mid && j <= r) {
        if (q[i] <= q[j]) tmp[k++] = q[i++];
        else {
            tmp[k++] = q[j++];
            res += mid - i + 1;
        }
    }
    while (i <= mid) tmp[k++] = q[i++];
    while (j <= r) tmp[k++] = q[j++];

    for (i = l, j = 0; i <= r; i++, j++) q[i] = tmp[j];
}

int main() {
    int n;
    cin >> n;
    for (int i = 1; i <= n; i++) cin >> a[i];
    for (int i = 1; i <= n; i++) {
        cin >> b[i];
        pos[b[i]] = i;  // 记录 B 中每个数的目标位置
    }

    // 把 A 映射成目标位置序列
    for (int i = 1; i <= n; i++) a[i] = pos[a[i]];

    // 求逆序对
    merge_sort(1, n, a);
    cout << res << endl;
    return 0;
}
```
