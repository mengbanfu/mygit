### 就你叫矩阵快速幂啊。

```c++
#include<iostream>
#include<cstring>
using namespace std;
const int N = 110;
typedef long long LL;
const LL mod = 1e9+7;
LL n; LL k;

struct mat{
    LL m[N][N]; //存矩阵的数组

    mat()
    {
        memset(m , 0 , sizeof m); //数组构造函数
    }

    //重载运算符
    mat operator* (const mat& B) const
    {
        mat C;
        for(int i = 1 ; i<=n ; i++)
            for(int j = 1 ; j<=n ; j++)
                for(int k = 1 ; k<=n ; k++)
                    C.m[i][j] = (C.m[i][j] + (m[i][k]*B.m[k][j])) % mod;
        return C;
    }
} A , RET;

void qpow(LL b)
{
    //初始化为单位矩阵
    for(int i = 1 ; i<= n ; i++) RET.m[i][i] = 1;
    
    while(b)
    {
        if(b&1)  RET = RET * A;
        A = A * A;
        b>>=1;
    }
}

int main()
{
    cin >> n >> k;
    for(int i = 1 ; i<=n ; i++)
        for(int j = 1 ; j<=n ; j++)
            cin >> A.m[i][j];

    qpow(k);

    for(int i = 1; i<=n ; i++)
    {
        for(int j = 1 ; j<=n ; j++)
        {
             cout << RET.m[i][j] << ' ';       
        }
        cout << endl;
    }
}
```
